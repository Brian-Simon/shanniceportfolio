/**
 * Font Setup Instructions
 * 
 * This folder should contain the font files used in the project:
 * - Space Grotesk (for headings)
 * - Inter (for body text)
 * 
 * Steps to add fonts:
 * 
 * 1. Download Space Grotesk from Google Fonts or Fontsource:
 *    - space-grotesk-bold.woff2
 *    - space-grotesk-semibold.woff2
 *    - space-grotesk-regular.woff2
 * 
 * 2. Download Inter from Google Fonts or Fontsource:
 *    - inter-regular.woff2
 *    - inter-medium.woff2
 *    - inter-semibold.woff2
 *    - inter-bold.woff2
 * 
 * 3. Place all font files in this app/fonts directory
 * 
 * Alternative: Use Google Fonts link in layout.tsx
 * Replace the localFont imports with:
 * 
 * import { Space_Grotesk, Inter } from 'next/font/google';
 * 
 * const spaceGrotesk = Space_Grotesk({ subsets: ['latin'] });
 * const inter = Inter({ subsets: ['latin'] });
 * 
 * Then in className: `${spaceGrotesk.className} ${inter.className}`
